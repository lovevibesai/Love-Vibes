export default {
  async scheduled(event: ScheduledEvent, env: Env) {
    // Runs every 6 hours via Cron Trigger
    
    // 1. Fetch user trust scores from D1
    const users = await env.DB.prepare(
      'SELECT id, trust_score FROM Users WHERE is_verified = 1'
    ).all();
    
    // 2. Batch update on-chain verification NFTs
    const updates = users.map(u => ({
      address: u.wallet_address,
      trust_score: u.trust_score
    }));
    
    // 3. Call Chainlink oracle to sync
    await fetch('https://chainlink-oracle.love-vibes.com/sync', {
      method: 'POST',
      body: JSON.stringify(updates)
    });
    
    // 4. Distribute staking rewards
    await distributeStakingRewards(env.DB);
  }
}

async function distributeStakingRewards(db: D1Database) {
  const activeStakes = await db.prepare(`
    SELECT * FROM staking 
    WHERE status = 'active' 
    AND end_date <= datetime('now')
  `).all();
  
  for (const stake of activeStakes) {
    const reward = calculateReward(stake.amount, stake.apy, stake.duration);
    
    await db.prepare(`
      UPDATE Users 
      SET credits_balance = credits_balance + ? 
      WHERE id = ?
    `).bind(reward, stake.user_id).run();
    
    await db.prepare(`
      UPDATE staking 
      SET status = 'completed' 
      WHERE id = ?
    `).bind(stake.id).run();
    
    // Log transaction
    await logLVCTransaction(db, stake.user_id, 'earn', reward, 'Staking reward');
  }
}

export async function checkDailyQuests(userId: string, env: Env) {
  const today = new Date().toISOString().split('T')[0];
  
  // Reset quests if new day
  await env.DB.prepare(`
    DELETE FROM daily_quests 
    WHERE user_id = ? AND reset_date < ?
  `).bind(userId, today).run();
  
  // Create today's quests if not exist
  const questTemplates = [
    { type: 'swipe_20', target: 20, lvc: 5, xp: 50 },
    { type: 'message_3', target: 3, lvc: 10, xp: 75 },
    { type: 'update_profile', target: 1, lvc: 8, xp: 60 }
  ];
  
  for (const quest of questTemplates) {
    await env.DB.prepare(`
      INSERT OR IGNORE INTO daily_quests 
      (id, user_id, quest_type, target, lvc_reward, xp_reward, reset_date)
      VALUES (?, ?, ?, ?, ?, ?, ?)
    `).bind(
      generateId(),
      userId,
      quest.type,
      quest.target,
      quest.lvc,
      quest.xp,
      today
    ).run();
  }
}

export async function updateQuestProgress(userId: string, questType: string, increment: number, env: Env) {
  const result = await env.DB.prepare(`
    UPDATE daily_quests 
    SET progress = progress + ? 
    WHERE user_id = ? AND quest_type = ? AND completed = FALSE
    RETURNING *
  `).bind(increment, userId, questType).first();
  
  if (result && result.progress >= result.target) {
    // Quest completed!
    await completeQuest(result.id, userId, env);
  }
}

async function completeQuest(questId: string, userId: string, env: Env) {
  const quest = await env.DB.prepare(`
    UPDATE daily_quests 
    SET completed = TRUE 
    WHERE id = ? 
    RETURNING lvc_reward, xp_reward
  `).bind(questId).first();
  
  // Award rewards
  await env.DB.prepare(`
    UPDATE Users 
    SET credits_balance = credits_balance + ? 
    WHERE id = ?
  `).bind(quest.lvc_reward, userId).run();
  
  await env.DB.prepare(`
    UPDATE user_progress 
    SET xp = xp + ? 
    WHERE user_id = ?
  `).bind(quest.xp_reward, userId).run();
  
  // Check level up
  await checkLevelUp(userId, env);
